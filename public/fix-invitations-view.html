招待ビューの修正

このツールは、invitations_viewを修正し、カラム名をsnake_caseに統一します。

実行するSQL

-- ビューを削除して再作成
DROP VIEW IF EXISTS invitations_v CASCADE;

-- 新しいビューを作成（カラム名をsnake_caseに統一）
CREATE OR REPLACE VIEW invitations_v AS
SELECT
    id,
    email,
    role,
    company_id,
    invite_token,  -- snake_case
    status,
    created_at,
    updated_at
FROM invitations;

-- RLSポリシーを設定
ALTER VIEW invitations_v ENABLE ROW LEVEL SECURITY;

-- 会社IDによる分離ポリシー
CREATE POLICY "company isolation" ON invitations_v
    USING (company_id = auth.jwt()->>'company_id');

-- 管理者は全ての招待を見ることができる
CREATE POLICY "admins can see all invitations" ON invitations_v
    USING (auth.jwt()->>'role' = '管理者');

-- RESTメタデータをリロード
SELECT pg_sleep(1);
SELECT rest_metadata_refresh();
